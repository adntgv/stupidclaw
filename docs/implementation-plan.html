# src/tools/base.py
from abc import ABC, abstractmethod
from dataclasses import dataclass

@dataclass
class ToolResult:
    success: bool
    output: str

class BaseTool(ABC):
    name: str = ""
    description: str = ""
    args_description: str = ""

    @abstractmethod
    def execute(self, args: str) -> ToolResult:
        ...

    def schema(self) -> str:
        return f"{self.name}: {self.description} | args: {self.args_description}"

# src/tools/web_search.py
from src.tools.base import BaseTool, ToolResult

class WebSearchTool(BaseTool):
    name = "web_search"
    description = "Search the web via DuckDuckGo. Returns top 5 results with title, URL, snippet."
    args_description = "search query string"

    def execute(self, args: str) -> ToolResult:
        try:
            from duckduckgo_search import DDGS
            results = []
            with DDGS() as ddgs:
                for r in ddgs.text(args, max_results=5):
                    results.append(f"[{r['title']}]({r['href']})\n{r['body']}")
            if not results:
                return ToolResult(True, "No results found.")
            return ToolResult(True, "\n\n".join(results))
        except Exception as e:
            return ToolResult(False, f"Search error: {e}")

import requests
from readability import Document
from bs4 import BeautifulSoup
from src.tools.base import BaseTool, ToolResult

class WebFetchTool(BaseTool):
    name = "web_fetch"
    description = "Fetch a URL and extract readable text content. Max 4000 chars."
    args_description = "URL to fetch"

    def execute(self, args: str) -> ToolResult:
        try:
            url = args.strip()
            resp = requests.get(url, timeout=10, headers={
                "User-Agent": "Mozilla/5.0 (compatible; StupidClaw/1.0)"
            })
            resp.raise_for_status()
            doc = Document(resp.text)
            soup = BeautifulSoup(doc.summary(), "html.parser")
            text = soup.get_text(separator="\n", strip=True)
            return ToolResult(True, text[:4000])
        except Exception as e:
            return ToolResult(False, f"Fetch error: {e}")

# src/tools/file_ops.py
import os
from pathlib import Path
from src.tools.base import BaseTool, ToolResult

SANDBOX = Path(os.environ.get("FILE_SANDBOX", "/app/data"))

def _safe_path(name: str) -> Path:
    """Resolve path inside sandbox, reject escapes."""
    p = (SANDBOX / name).resolve()
    if not str(p).startswith(str(SANDBOX.resolve())):
        raise ValueError("Path escapes sandbox")
    return p

class FileReadTool(BaseTool):
    name = "file_read"
    description = "Read a file from the data directory. Returns content (max 4000 chars)."
    args_description = "relative filename (e.g. notes.txt)"

    def execute(self, args: str) -> ToolResult:
        try:
            p = _safe_path(args.strip())
            if not p.exists():
                return ToolResult(False, "File not found")
            return ToolResult(True, p.read_text()[:4000])
        except Exception as e:
            return ToolResult(False, f"Read error: {e}")

class FileWriteTool(BaseTool):
    name = "file_write"
    description = "Write content to a file in the data directory. Creates dirs as needed."
    args_description = "filename|||content (triple pipe separator)"

    def execute(self, args: str) -> ToolResult:
        try:
            parts = args.split("|||", 1)
            if len(parts) != 2:
                return ToolResult(False, "Format: filename|||content")
            p = _safe_path(parts[0].strip())
            p.parent.mkdir(parents=True, exist_ok=True)
            p.write_text(parts[1])
            return ToolResult(True, f"Written {len(parts[1])} chars to {parts[0].strip()}")
        except Exception as e:
            return ToolResult(False, f"Write error: {e}")

class FileListTool(BaseTool):
    name = "file_list"
    description = "List files in the data directory."
    args_description = "subdirectory (optional, default root)"

    def execute(self, args: str) -> ToolResult:
        try:
            p = _safe_path(args.strip() or ".")
            if not p.is_dir():
                return ToolResult(False, "Not a directory")
            files = [str(f.relative_to(SANDBOX)) for f in sorted(p.rglob("*")) if f.is_file()]
            return ToolResult(True, "\n".join(files[:50]) or "(empty)")
        except Exception as e:
            return ToolResult(False, f"List error: {e}")
src/tools/shell.py — Sandboxed shell execution with allowlist
# src/tools/shell.py
import subprocess
import shlex
from src.tools.base import BaseTool, ToolResult

ALLOWED_COMMANDS = {
    "ls", "cat", "head", "tail", "wc", "grep",
    "date", "whoami", "uname", "df", "du",
    "echo", "sort", "uniq", "cut", "tr",
    "python3", "pip", "curl",
}

class ShellTool(BaseTool):
    name = "shell"
    description = "Run a shell command (allowlisted: ls, cat, head, tail, grep, python3, curl, etc). Max 10s timeout."
    args_description = "shell command string"

    def execute(self, args: str) -> ToolResult:
        try:
            parts = shlex.split(args.strip())
            if not parts:
                return ToolResult(False, "Empty command")
            cmd = parts[0].split("/")[-1]  # basename
            if cmd not in ALLOWED_COMMANDS:
                return ToolResult(False, f"Command '{cmd}' not in allowlist: {sorted(ALLOWED_COMMANDS)}")
            result = subprocess.run(
                args.strip(), shell=True, capture_output=True,
                text=True, timeout=10, cwd="/app/data"
            )
            output = (result.stdout + result.stderr).strip()
            return ToolResult(result.returncode == 0, output[:4000] or "(no output)")
        except subprocess.TimeoutExpired:
            return ToolResult(False, "Command timed out (10s limit)")
        except Exception as e:
            return ToolResult(False, f"Shell error: {e}")

src/memory/session.py — Sliding window with token counting
# src/memory/session.py
from dataclasses import dataclass, field
from typing import List, Dict

def count_tokens(text: str) -> int:
    """Rough token count: ~4 chars per token for English."""
    return len(text) // 4

@dataclass
class SessionMemory:
    max_tokens: int = 6000
    messages: List[Dict] = field(default_factory=list)

    def add(self, role: str, content: str):
        self.messages.append({"role": role, "content": content})
        self._trim()

    def _trim(self):
        """Slide window: drop oldest messages (keep system) until under limit."""
        while self._total_tokens() > self.max_tokens and len(self.messages) > 2:
            # Find first non-system message to drop
            for i, m in enumerate(self.messages):
                if m["role"] != "system":
                    self.messages.pop(i)
                    break

    def _total_tokens(self) -> int:
        return sum(count_tokens(m["content"]) for m in self.messages)

    def get_messages(self) -> List[Dict]:
        return list(self.messages)

    def get_summary(self) -> str:
        """One-line summary for pull-based context index."""
        n = len([m for m in self.messages if m["role"] == "user"])
        if not self.messages:
            return "(no conversation yet)"
        last = self.messages[-1]["content"][:80]
        return f"{n} user messages. Last: {last}..."

    def clear(self):
        self.messages.clear()

# src/memory/semantic.py
import chromadb
from chromadb.utils import embedding_functions
import logging

logger = logging.getLogger(__name__)

class SemanticMemory:
    def __init__(self, persist_dir: str = "/app/data/chromadb",
                 model: str = "all-MiniLM-L6-v2"):
        self.ef = embedding_functions.SentenceTransformerEmbeddingFunction(model_name=model)
        self.client = chromadb.PersistentClient(path=persist_dir)
        self.collection = self.client.get_or_create_collection(
            name="memory", embedding_function=self.ef
        )

    def store(self, text: str, metadata: dict = None, doc_id: str = None):
        import uuid
        doc_id = doc_id or str(uuid.uuid4())
        self.collection.upsert(
            ids=[doc_id],
            documents=[text],
            metadatas=[metadata or {}]
        )

    def search(self, query: str, n: int = 5) -> list:
        results = self.collection.query(query_texts=[query], n_results=n)
        out = []
        for i, doc in enumerate(results["documents"][0]):
            out.append({
                "id": results["ids"][0][i],
                "text": doc,
                "distance": results["distances"][0][i],
                "metadata": results["metadatas"][0][i],
            })
        return out

    def count(self) -> int:
        return self.collection.count()

# src/memory/episodic.py
import json
from datetime import datetime
from pathlib import Path

class EpisodicMemory:
    def __init__(self, base_dir: str = "/app/data/memory/episodes"):
        self.base = Path(base_dir)
        self.base.mkdir(parents=True, exist_ok=True)

    def log(self, user_msg: str, bot_reply: str, route: str,
            tools_used: list = None, verified: bool = True):
        today = datetime.utcnow().strftime("%Y-%m-%d")
        entry = {
            "ts": datetime.utcnow().isoformat(),
            "user": user_msg[:500],
            "reply": bot_reply[:500],
            "route": route,
            "tools": tools_used or [],
            "verified": verified,
        }
        f = self.base / f"{today}.jsonl"
        with open(f, "a") as fp:
            fp.write(json.dumps(entry) + "\n")

    def get_today(self) -> list:
        today = datetime.utcnow().strftime("%Y-%m-%d")
        f = self.base / f"{today}.jsonl"
        if not f.exists():
            return []
        return [json.loads(line) for line in f.read_text().strip().split("\n") if line]

    def get_summary(self) -> str:
        entries = self.get_today()
        if not entries:
            return "No conversations today."
        return f"{len(entries)} exchanges today. Topics: " + \
               ", ".join(e["user"][:40] for e in entries[-3:])

from pathlib import Path

class ProceduralMemory:
    def __init__(self, base_dir: str = "/app/data/memory"):
        self.base = Path(base_dir)
        self.base.mkdir(parents=True, exist_ok=True)
        self.files = {
            "soul": self.base / "SOUL.md",
            "user": self.base / "USER.md",
            "memory": self.base / "MEMORY.md",
        }
        # Create defaults if missing
        for key, path in self.files.items():
            if not path.exists():
                path.write_text(f"# {key.upper()}\n\n(empty)\n")

    def read(self, name: str) -> str:
        path = self.files.get(name)
        if path and path.exists():
            return path.read_text()[:2000]
        return ""

    def update(self, name: str, content: str):
        path = self.files.get(name)
        if path:
            path.write_text(content)

    def get_identity_prompt(self) -> str:
        """Compact identity for system prompt injection."""
        soul = self.read("soul")
        user = self.read("user")
        parts = []
        if soul:
            parts.append(f"[Identity] {soul[:500]}")
        if user:
            parts.append(f"[User] {user[:500]}")
        return "\n".join(parts)
src/memory/manager.py — Unified memory with pull-based context
# src/memory/manager.py
from src.memory.session import SessionMemory
from src.memory.semantic import SemanticMemory
from src.memory.episodic import EpisodicMemory
from src.memory.procedural import ProceduralMemory
import logging

logger = logging.getLogger(__name__)

class MemoryManager:
    def __init__(self):
        self.sessions: dict[str, SessionMemory] = {}
        self.semantic = SemanticMemory()
        self.episodic = EpisodicMemory()
        self.procedural = ProceduralMemory()

    def get_session(self, user_id: str) -> SessionMemory:
        if user_id not in self.sessions:
            self.sessions[user_id] = SessionMemory()
        return self.sessions[user_id]

    def get_context_index(self, user_id: str) -> str:
        """Pull-based: tiny index the model sees. It can request chunks by ID."""
        session = self.get_session(user_id)
        return (
            f"[session] {session.get_summary()}\n"
            f"[episodic] {self.episodic.get_summary()}\n"
            f"[semantic] {self.semantic.count()} stored memories\n"
            f"[procedural] SOUL.md, USER.md, MEMORY.md available\n"
            f"To retrieve: use tool 'memory_recall' with a query."
        )

    def recall(self, query: str, n: int = 3) -> str:
        """Semantic search across stored memories."""
        results = self.semantic.search(query, n)
        if not results:
            return "No relevant memories found."
        return "\n---\n".join(
            f"[{r['id'][:8]}] (dist={r['distance']:.2f}) {r['text']}"
            for r in results
        )

    def store_exchange(self, user_id: str, user_msg: str,
                       bot_reply: str, route: str, tools: list = None,
                       verified: bool = True):
        # Session
        session = self.get_session(user_id)
        session.add("user", user_msg)
        session.add("assistant", bot_reply)
        # Episodic
        self.episodic.log(user_msg, bot_reply, route, tools, verified)
        # Semantic (only store if substantial)
        if len(user_msg) > 20:
            self.semantic.store(
                f"User: {user_msg}\nBot: {bot_reply[:200]}",
                metadata={"user_id": user_id, "route": route}
            )

# src/registry.py
import importlib
import logging
from pathlib import Path
from src.tools.base import BaseTool

logger = logging.getLogger(__name__)

class ToolRegistry:
    def __init__(self):
        self._tools: dict[str, BaseTool] = {}

    def register(self, tool: BaseTool):
        self._tools[tool.name] = tool
        logger.info(f"Registered tool: {tool.name}")

    def get(self, name: str) -> BaseTool | None:
        return self._tools.get(name)

    def list_tools(self) -> list[str]:
        return list(self._tools.keys())

    def get_schemas(self) -> str:
        """Compact schema string for LLM prompts."""
        return "\n".join(t.schema() for t in self._tools.values())

    def match(self, query: str) -> list[str]:
        """Simple keyword matching for tool selection."""
        query_lower = query.lower()
        matches = []
        for name, tool in self._tools.items():
            if any(w in query_lower for w in [name] + tool.description.lower().split()[:5]):
                matches.append(name)
        return matches

    def execute(self, name: str, args: str):
        tool = self.get(name)
        if not tool:
            from src.tools.base import ToolResult
            return ToolResult(False, f"Unknown tool: {name}. Available: {self.list_tools()}")
        return tool.execute(args)

# src/skills.py
from pathlib import Path
import re
import logging

logger = logging.getLogger(__name__)

class Skill:
    def __init__(self, path: Path):
        self.path = path
        self.name = path.parent.name
        content = path.read_text()
        # Extract description from first paragraph after # header
        lines = content.split("\n")
        self.description = ""
        for line in lines[1:]:
            if line.strip() and not line.startswith("#"):
                self.description = line.strip()
                break
        self.content = content

class SkillManager:
    def __init__(self, skills_dir: str = "/app/data/skills"):
        self.skills_dir = Path(skills_dir)
        self.skills: dict[str, Skill] = {}
        self._load_skills()

    def _load_skills(self):
        if not self.skills_dir.exists():
            self.skills_dir.mkdir(parents=True, exist_ok=True)
            return
        for skill_md in self.skills_dir.rglob("SKILL.md"):
            try:
                skill = Skill(skill_md)
                self.skills[skill.name] = skill
                logger.info(f"Loaded skill: {skill.name}")
            except Exception as e:
                logger.error(f"Failed to load skill {skill_md}: {e}")

    def match(self, query: str) -> Skill | None:
        """Simple keyword match. Returns best matching skill or None."""
        query_lower = query.lower()
        best = None
        best_score = 0
        for name, skill in self.skills.items():
            score = sum(1 for w in query_lower.split()
                        if w in name.lower() or w in skill.description.lower())
            if score > best_score:
                best_score = score
                best = skill
        return best if best_score > 0 else None

    def get_index(self) -> str:
        """Compact index for pull-based context."""
        if not self.skills:
            return "(no skills installed)"
        return "\n".join(f"- {s.name}: {s.description[:80]}" for s in self.skills.values())

from apscheduler.schedulers.asyncio import AsyncIOScheduler
from apscheduler.triggers.cron import CronTrigger
from apscheduler.triggers.interval import IntervalTrigger
import logging

logger = logging.getLogger(__name__)

class TaskScheduler:
    def __init__(self):
        self.scheduler = AsyncIOScheduler()
        self._tasks = {}

    def start(self):
        self.scheduler.start()
        logger.info("Scheduler started")

    def add_interval(self, name: str, func, minutes: int = 30):
        job = self.scheduler.add_job(func, IntervalTrigger(minutes=minutes), id=name)
        self._tasks[name] = job

    def add_cron(self, name: str, func, hour: int, minute: int = 0):
        job = self.scheduler.add_job(func, CronTrigger(hour=hour, minute=minute), id=name)
        self._tasks[name] = job

    def remove(self, name: str):
        if name in self._tasks:
            self._tasks[name].remove()
            del self._tasks[name]

    def list_tasks(self) -> list:
        return [{"name": k, "next_run": str(v.next_run_time)} for k, v in self._tasks.items()]

# src/evidence.py
import logging
from src.tools.base import ToolResult

logger = logging.getLogger(__name__)

class EvidenceLoop:
    """
    ECoRAG-style: retrieve → ask "is this enough?" → retrieve more or answer.
    Max 3 iterations to bound cost.
    """
    def __init__(self, llm_fn, search_fn, max_rounds: int = 3):
        self.llm = llm_fn
        self.search = search_fn
        self.max_rounds = max_rounds

    def gather(self, question: str) -> str:
        evidence = []
        query = question

        for i in range(self.max_rounds):
            # Retrieve
            results = self.search(query)
            evidence.append(results)

            # Sufficiency check
            evidence_text = "\n---\n".join(evidence)
            verdict = self.llm([
                {"role": "system", "content": (
                    "You are an evidence evaluator. Given a question and evidence, reply:\n"
                    "SUFFICIENT - if the evidence is enough to answer confidently\n"
                    "NEED_MORE:  - if more info is needed\n"
                    "Reply with ONLY one of these formats."
                )},
                {"role": "user", "content": f"Question: {question}\nEvidence:\n{evidence_text}"}
            ])

            if "SUFFICIENT" in verdict.upper():
                logger.info(f"Evidence sufficient after {i+1} rounds")
                break
            elif "NEED_MORE" in verdict.upper():
                query = verdict.split(":", 1)[-1].strip()
                logger.info(f"Round {i+1}: need more, refined query: {query}")
            else:
                break

        return "\n---\n".join(evidence)

# src/prompt_optimizer.py
import json
from pathlib import Path
from datetime import datetime

class PromptOptimizer:
    """
    Workflow:
    1. Maintain eval_set.jsonl: {input, expected_output, tags}
    2. Run all evals against current prompts
    3. Score with LLM judge (0-10)
    4. Try prompt variants, keep winners
    """

    def __init__(self, eval_path: str = "/app/data/evals/eval_set.jsonl",
                 results_dir: str = "/app/data/evals/results"):
        self.eval_path = Path(eval_path)
        self.results_dir = Path(results_dir)
        self.results_dir.mkdir(parents=True, exist_ok=True)

    def load_evals(self) -> list:
        if not self.eval_path.exists():
            return []
        return [json.loads(l) for l in self.eval_path.read_text().strip().split("\n") if l]

    def run_eval(self, agent_fn, llm_judge_fn) -> dict:
        """Run all evals, score with LLM judge, save results."""
        evals = self.load_evals()
        results = []
        total_score = 0

        for ev in evals:
            actual = agent_fn(ev["input"])
            score_resp = llm_judge_fn([
                {"role": "system", "content": "Score 0-10 how well the actual answer matches expected. Reply ONLY with the number."},
                {"role": "user", "content": f"Expected: {ev['expected_output']}\nActual: {actual}"}
            ])
            try:
                score = int(score_resp.strip())
            except:
                score = 5
            results.append({**ev, "actual": actual, "score": score})
            total_score += score

        avg = total_score / len(results) if results else 0
        report = {
            "timestamp": datetime.utcnow().isoformat(),
            "avg_score": round(avg, 2),
            "total": len(results),
            "results": results,
        }
        out = self.results_dir / f"eval_{datetime.utcnow().strftime('%Y%m%d_%H%M%S')}.json"
        out.write_text(json.dumps(report, indent=2))
        return report

import asyncio
import json
import websockets
import logging

logger = logging.getLogger(__name__)

class GatewayClient:
    """Connect to OpenClaw gateway to reuse its channels (Telegram, Discord, etc)."""

    def __init__(self, url: str, password: str):
        self.url = url
        self.password = password
        self.ws = None

    async def connect(self):
        self.ws = await websockets.connect(self.url)
        # Authenticate
        await self.ws.send(json.dumps({
            "type": "auth",
            "password": self.password,
        }))
        resp = json.loads(await self.ws.recv())
        if resp.get("type") != "auth_ok":
            raise ConnectionError(f"Gateway auth failed: {resp}")
        logger.info("Connected to OpenClaw gateway")

    async def send_message(self, channel: str, target: str, text: str):
        await self.ws.send(json.dumps({
            "type": "message",
            "action": "send",
            "channel": channel,
            "target": target,
            "message": text,
        }))

    async def listen(self, handler):
        async for msg in self.ws:
            data = json.loads(msg)
            if data.get("type") == "message":
                await handler(data)

# src/subagents.py
import asyncio
import logging
from dataclasses import dataclass
from typing import Callable

logger = logging.getLogger(__name__)

@dataclass
class SubTask:
    description: str
    handler: Callable
    args: dict

class SubAgentManager:
    def __init__(self, agent):
        self.agent = agent
        self._running = {}

    def decompose(self, task: str) -> list[SubTask]:
        """Use LLM to break a complex task into subtasks."""
        resp = self.agent._llm([
            {"role": "system", "content": (
                "Break this task into 2-4 independent subtasks. "
                'Reply JSON: [{"description":"...","tool":"web_search","args":"..."}]'
            )},
            {"role": "user", "content": task}
        ])
        parsed = self.agent._extract_json(resp)
        if isinstance(parsed, list):
            return [SubTask(s.get("description", ""), self.agent._execute_tool, s) for s in parsed]
        return []

    async def run_parallel(self, subtasks: list[SubTask]) -> list[str]:
        """Run subtasks concurrently."""
        loop = asyncio.get_event_loop()
        tasks = [
            loop.run_in_executor(None, st.handler, st.args.get("tool", "none"), st.args.get("args", ""))
            for st in subtasks
        ]
        return await asyncio.gather(*tasks)

from src.tools.base import BaseTool, ToolResult

class BrowserTool(BaseTool):
    name = "browser"
    description = "Open a URL and extract visible text content using headless Playwright."
    args_description = "URL to visit"

    def execute(self, args: str) -> ToolResult:
        try:
            from playwright.sync_api import sync_playwright
            with sync_playwright() as p:
                browser = p.chromium.launch(headless=True)
                page = browser.new_page()
                page.goto(args.strip(), timeout=15000)
                page.wait_for_load_state("domcontentloaded")
                text = page.inner_text("body")
                browser.close()
                return ToolResult(True, text[:4000])
        except Exception as e:
            return ToolResult(False, f"Browser error: {e}")
